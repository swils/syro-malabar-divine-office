\PSALMtitle{86}{Bede om hulp}

\psalmsubtitle{a) Mijn God, help mij!}

\begin{halfparskip}
  \psalmverse{Neig Uw oor, O Heer; verhoor mij,~\sep\ want ik ben ellendig en arm.}

  \qanona{Christus, de Vriend van de boeteling, open de deur voor ons gebed en aanvaard onze bede.}

  \psalmverse{Bescherm mij, want ik ben U toegewijd;~\sep\ red Uw dienaar, die op U hoopt.}

  \psalmverse{Mijn God zijt Gij; wees mij genadig, O Heer,~\sep\ want almaar door roep ik tot U.}

  \psalmverse{Verblijd de ziel van Uw dienaar,~\sep\ want tot U, O Heer, verhef ik mijn ziel.}

  \psalmverse{Want Gij, O Heer, zijt goed en genadig,~\sep\ vol erbarming voor al wie U aanroept.}

  \psalmverse{Luister, Heer, naar mijn bede,~\sep\ en geef acht op de stem van mijn smeken.}

  \psalmverse{Op de dag van mijn kwelling riep ik tot U,~\sep\ omdat Gij mij verhoren zult.}
\end{halfparskip}

\psalmsubtitle{b) Gij zult mij helpen, O Heer}

\begin{halfparskip}
  \psalmverse{Onder de goden, O Heer, is er geen als Gij,~\sep\ en geen werk is gelijk aan het Uwe.}

  \psalmverse{Alle volken, door U geschapen, zullen komen, en U aanbidden, O Heer,~\sep\ en verheerlijken Uw Naam.}

  \psalmverse{Want Gij zijt groot en Gij doet wonderwerken:~\sep\ Gij zijt God, en Gij alleen.}
\end{halfparskip}

\psalmsubtitle{c) Gij wilt mij helpen, O Heer}

\begin{halfparskip}
  \psalmverse{Toon mij Uw weg, O Heer, opdat ik wandele in Uw waarheid,~\sep\ richt mijn hart op de vrees voor Uw Naam.}

  \psalmverse{Ik zal U prijzen, Heer, mijn God, uit heel mijn hart:~\sep\ en eeuwig Uw Naam verheerlijken.}

  \psalmverse{Want Uw erbarming voor mij was groot,~\sep\ en uit de diepten van het dodenrijk hebt Gij mij opgehaald.}

  \psalmverse{Trotsen, O God, zijn tegen mij opgestaan, een bende geweldenaars staat mij naar het leven,~\sep\ zij houden U niet voor ogen.}

  \psalmverse{Maar Gij, O Heer, zijt een barmhartige en liefdevolle God,~\sep\ lankmoedig, rijk aan ontferming en trouw.}

  \psalmverse{Blik op mij neer en wees mij genadig;~\sep\ schenk aan Uw dienaar Uw kracht, en red de zoon van Uw dienstmaagd.}

  \psalmverse{Geef mij een teken van Uw gunst, opdat die mij haten, Heer, vol schaamte zien,~\sep\ dat Gij, O Heer, mij hulp en troost hebt geschonken.}
\end{halfparskip}
