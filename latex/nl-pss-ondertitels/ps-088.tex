\PSALMtitle{88}{De vertwijfeling nabij!}

\psalmsubtitle{a) Groot, Heer, is mijn ellende!}

\begin{halfparskip}
  \psalmverse{Heer, mijn God, ik roep overdag,~\sep\ en ik jammer 's nachts voor Uw aanschijn.}

  \qanona{Gij, die ons gemaakt hebt, zijt barmhartig; in Uw goedertierenheid heb medelijden met ons.}

  \psalmverse{Dringe mijn bede toch door tot U,~\sep\ neig Uw oor naar mijn klagen!}

  \psalmverse{Want mijn ziel is verzadigd met rampen,~\sep\ mijn leven is het dodenrijk nabij.}

  \psalmverse{Ik word gerekend onder hen, die ten grave dalen,~\sep\ ik ben als een man zonder kracht.}

  \psalmverse{Onder de doden is mijn legerstede,~\sep\ als van verslagenen, die liggen in het graf,}

  \psalmverse{Aan wie Gij niet meer denkt,~\sep\ die aan Uw zorgen zijn onttrokken.}

  \psalmverse{In een diepe groeve hebt Gij mij neergelegd,~\sep\ in duisternis, in een diep ravijn.}

  \psalmverse{Uw verontwaardiging drukt zwaar op mij,~\sep\ met al Uw golven slaat Gij mij neer.}

  \psalmverse{Gij hebt mijn vrienden van mij vervreemd, mij tot afschuw voor hen gemaakt;~\sep\ ik zit gevangen, en kan niet ontkomen.}
\end{halfparskip}

\psalmsubtitle{b) Heer, kom mij te hulp!}

\begin{halfparskip}
  \psalmverse{Van ellende verkwijnen mijn ogen; iedere dag roep ik tot U, O Heer,~\sep\ naar U strek ik mijn handen uit.}

  \psalmverse{Of doet Gij voor doden nog wonderen,~\sep\ of zullen gestorvenen, herrijzend, U loven?}

  \psalmverse{Of wordt Uw goedheid in het graf verkondigd,~\sep\ Uw trouw in het dodenrijk?}

  \psalmverse{Openbaart men in het duister Uw wonderen,~\sep\ in het land der vergetelheid Uw genade?}
\end{halfparskip}

\psalmsubtitle{c) Help mij, Heer, in mijn ongeluk!}

\begin{halfparskip}
  \psalmverse{Ik echter roep tot U, O Heer,~\sep\ mijn bede stijgt tot U op in de morgen.}

  \psalmverse{Waarom toch, O Heer, verstoot Gij mij,~\sep\ verbergt Gij voor mij Uw gelaat?}

  \psalmverse{Van jongs af ben ik ellendig en stervend,~\sep\ ik torste Uw verschrikkingen en kwijnde.}

  \psalmverse{Uw toorn is over mij heengegaan,~\sep\ Uw verschrikkingen sloegen mij neer.}

  \psalmverse{Zij omgeven mij immer als water,~\sep\ omringen mij alle tezamen.}

  \psalmverse{Vriend en makker hebt Gij van mij vervreemd,~\sep\ mijn vertrouweling is de duisternis.}
\end{halfparskip}
