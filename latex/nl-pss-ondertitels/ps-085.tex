\PSALMtitle{85}{Verlangen naar het Messiaanse Rijk}

\psalmsubtitle{a) Dank, Heer, voor de bevrijding!}

\begin{halfparskip}
  \psalmverse{Gij zijt Uw land genadig geweest, O Heer,~\sep\ hebt het lot van Jacob ten goede gekeerd.}

  \qanona{Zend, O onze Heer, hulp en redding aan Uw gelovigen door de grote kracht van Uw Kruis.}

  \liturgicalhint{Of:} \emph{Wie de zon wil afbeelden, vergist zich enorm, want hij heeft niet het verstand om haar heerlijke dingen te vergelijken (\translationoptionNl{te bevatten}). Hij die een erfgenaam wil zijn van het hemels koninkrijk, moet zijn ziel bevrijden van de slavernij van deze wereld.}

  \psalmverse{Vergeven hebt Gij de schuld van Uw volk,~\sep\ en al zijn zonden bedekt.}

  \psalmverse{Uw gramschap hebt Gij geheel bedwongen,~\sep\ de gloed van Uw toorn gestild.}
\end{halfparskip}

\psalmsubtitle{b) Voltooi, Heer, de Verlossing!}

\begin{halfparskip}
  \psalmverse{Herstel ons, o God, onze Redder,~\sep\ en leg Uw wrevel tegen ons af.}

  \psalmverse{Zult Gij dan eeuwig tegen ons toornen,~\sep\ of verbolgen blijven van geslacht tot geslacht?}

  \psalmverse{Zult Gij ons dan niet opnieuw doen leven,~\sep\ opdat Uw volk zich verblijde in U?}

  \psalmverse{Toon ons, Heer, Uw barmhartigheid,~\sep\ en schenk ons Uw heil!}
\end{halfparskip}

\psalmsubtitle{c) Gij zult ons zegenen, Heer}

\begin{halfparskip}
  \psalmverse{Ik wil horen naar wat de Heer God spreekt:~\sep\ vrede voorzeker kondigt Hij aan.}

  \psalmverse{Voor Zijn volk en Zijn heiligen,~\sep\ en voor hen, die zich van harte keren tot Hem.}

  \psalmverse{Ja waarlijk, Zijn heil is nabij voor wie Hem vrezen,~\sep\ en zo zal er glorie wonen in ons land:}

  \psalmverse{Barmhartigheid en trouw zullen elkander ontmoeten,~\sep\ gerechtigheid en vrede elkander de kus  geven.}

  \psalmverse{Trouw zal aan de aarde ontspruiten,~\sep\ en gerechtigheid neerzien vanuit de hemel.}

  \psalmverse{De Heer zelf zal zegen schenken,~\sep\ en ons land zijn vruchten geven.}

  \psalmverse{Gerechtigheid zal vóór Hem uitgaan,~\sep\ en heil zijn schreden volgen.}
\end{halfparskip}
