\PSALMtitle{84}{Verlangen naar het huis van God}

\psalmsubtitle{a) Mijn God, ik verlang naar Uw heiligdom}

\begin{halfparskip}
  \psalmverse{Hoe liefelijk is Uw woonstede, Heer der legerscharen;~\sep\ mijn ziel verlangt, ziet smachtend uit naar de voorhoven van de Heer.}

  \qanona{Hoe heerlijk en prachtig is Uw heiligdom, O God, die alles heiligt.}

  \psalmverse{Mijn hart en mijn lichaam,~\sep\ juichen voor de levende God.}

  \psalmverse{Ook de mus vindt een woning,~\sep\ en de zwaluw een nest, waar ze haar jongen in neerlegt.}

  \psalmverse{Uw altaren, O Heer der legerscharen,~\sep\ mijn Koning en mijn God!}
\end{halfparskip}

\psalmsubtitle{b) Gelukkig hij, die naar het heiligdom gaat}

\begin{halfparskip}
  \psalmverse{Gelukkig zij, die wonen in Uw huis, O Heer;~\sep\ eeuwig loven zij U.}

  \psalmverse{Gelukkig de man, die hulp krijgt van U,~\sep\ als hij het plan heeft op bedevaart te gaan:}

  \psalmverse{Trekken zij door een dorre vallei, dan maken zij haar tot bron,~\sep\ en de vroege regen bekleedt haar met zegeningen.}

  \psalmverse{Al gaande zal hun kracht vermeerderen:~\sep\ de God der goden zullen zij in Sion zien.}
\end{halfparskip}

\psalmsubtitle{c) Leid mij, Heer, naar Uw heiligdom!}

\begin{halfparskip}
  \psalmverse{Heer der legerscharen, hoor naar mijn bede,~\sep\ ach, luister toch, O God van Jacob.}

  \psalmverse{Zie toe, O God, ons schild,~\sep\ en zie op het gelaat van Uw gezalfde.}

  \psalmverse{Waarlijk, één dag in Uw voorhoven is beter,~\sep\ dan duizend andere.}

  \psalmverse{Liever blijf ik staan op de drempel van het huis van mijn God,~\sep\ dan te toeven in de tenten der bozen.}

  \psalmverse{Want een zon en een schild is God de Heer;~\sep\ de Heer schenkt genade en glorie.}

  \psalmverse{Hij weigert het goede niet,~\sep\ aan die in onschuld wandelen.}

  \psalmverse{Heer der legerscharen,~\sep\ gelukkig de mens, die op U vertrouwt.}
\end{halfparskip}
