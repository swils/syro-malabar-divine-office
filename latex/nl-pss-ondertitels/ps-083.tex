\PSALMtitle{83}{De vijand dringt op!}

\psalmsubtitle{a) De vijanden, Heer, spannen tegen ons samen}

\begin{halfparskip}
  \psalmverse{Wil niet zwijgen, O Heer,~\sep\ wil niet zwijgen, O God, noch rusten!}

  \qanona{Er is niemand zoals U onder de dingen die gemaakt zijn, O almachtige God.}

  \psalmverse{Want zie, Uw vijanden woelen,~\sep\ en die U haten, steken het hoofd omhoog.}

  \psalmverse{Tegen Uw volk smeden zij plannen,~\sep\ en tegen Uw beschermelingen spannen zij samen.}

  \psalmverse{``Komt'', zo spreken zij, ``vagen wij hen weg uit de rij van de volken,~\sep\ en dat men de naam van Israël niet meer gedenke.''}

  \psalmverse{Waarlijk, zij overleggen eensgezind,~\sep\ en sluiten een verbond tegen U:}

  \psalmverse{De tenten van Edom en de Ismaëlieten,~\sep\ Moab en de zonen van Hagar,}

  \psalmverse{Gebal, Ammon en Amalec,~\sep\ de Filistijnen en de bewoners van Tyrus;}

  \psalmverse{Ook de Assyriërs verbonden zich met hen,~\sep\ en leenden hun arm aan de zonen van Lot.}
\end{halfparskip}

\psalmsubtitle{b) Verdelg de vijand, O God, gelijk weleer!}

\begin{halfparskip}
  \psalmverse{Behandel hen als Madian,~\sep\ als Sisara en Jabin bij de beek Cison,}

  \psalmverse{Die bij Endor werden verdelgd,~\sep\ en tot mest op het veld zijn gemaakt.}

  \psalmverse{Maak hun vorsten als Oreb en Zeb,~\sep\ als Zebëe en Salmana, al hun leiders,}

  \psalmverse{Die zeiden:~\sep\ ``Laten wij het gebied van God gaan bezetten.''}

  \psalmverse{Maak ze, mijn God, als bladeren die dwarrelen in de storm,~\sep\ als een strohalm, heen en weer gezwiept door de wind,}

  \psalmverse{Als vuur, dat bossen verteert,~\sep\ en als een vlam, die bergen verschroeit,}

  \psalmverse{Zo moogt Gij hen vervolgen met Uw stormwind,~\sep\ door Uw orkaan hen verwarren.}

  \psalmverse{Overdek hun gelaat met schande,~\sep\ opdat zij Uw Naam zoeken, O Heer;}

  \psalmverse{Laat ze beschaamd en verbijsterd staan voor eeuwig,~\sep\ laat ze te schande worden en vergaan.}

  \psalmverse{Dat zij erkennen dat Gij, wiens naam de Heer is,~\sep\ de enig Verhevene zijt over heel de aarde.}
\end{halfparskip}
